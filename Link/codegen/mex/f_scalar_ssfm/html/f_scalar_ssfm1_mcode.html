1function [ sig ] = f_scalar_ssfm( ch,Pavg,sig )
 2    %SCALAR_SSFM Summary of this function goes here
 3    %   Detailed explanation goes here
 4
 5    omega = 2*pi*sig.SYMBOLRATE*sig.FN*1e9;            % [rad/s]
 6    beta  = 0.5*omega.^2*ch.b2 + omega.^3*ch.b3/6;
 7
 8    ux    = sig.FIELDX;%;get(sig,'FIELDX');
 9
10    if abs(ch.alphalin*ch.dz) > 1e-6
11        Leff     = (1-exp(-ch.alphalin*ch.dz))/ch.alphalin;
12    else
13        Leff     = ch.dz;
14    end
15    steps = (0:ch.nstep-1);
16    z    = ch.dz * steps;
17
18    xi   = ch.gamma*Leff*exp(-ch.alphalin*z)*Pavg;
19
20    halfdz      = ch.dz/2;
21
22    
23    %                           HALF DZ GVD                  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%    %
24    %                                DZ SPM                      %
25    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
26    ux = scalar_lin_step(beta*halfdz,ux);
27    ux = scalar_nl_step(ux,xi(1));
28    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
29
30    for i=2:ch.nstep
31
32        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
33        %                           DZ GVD                       %
34        %                           DZ SPM                       %
35        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
36        ux = scalar_lin_step(beta*ch.dz,ux);
37        ux = scalar_nl_step(ux,xi(i));
38        %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
39
40    end
41
42    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
43    %                      LAST HALF DZ GVD                      %
44    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
45    ux = scalar_lin_step(beta*halfdz,ux);
46    %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
47
48    sig.FIELDX = ux;%set(sig,'FIELDX',ux);
49
50
51end
52

53function ux  = scalar_lin_step(betaxdz,ux)
54
55    Hf = exp(-1i*betaxdz);
56    ux = ifft( fft(ux) .* Hf);
57
58end
59
60function ux  = scalar_nl_step(ux,xi)
61
62    pow = real(ux).^2 + imag(ux).^2;
63    ux  = ux.*exp(-1i*xi.*pow);
64
65end
